                    #include <stdio.h>
                    int main() 
                    {
                    FILE *fp;
                    char name[50], address[100];
                    int age;
                    float salary;
                    // Create a new file
                    fp = fopen("STAFF.TXT", "a");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    return 1;
                    }
                    // Input employee data
                    for (int i = 0; i < 2; i++) {
                    printf("Enter employee %d's name: ", i + 1);
                    scanf("%s", name);
                    printf("Enter employee %d's address: ", i + 1);
                    scanf("%s", address);
                    printf("Enter employee %d's age: ", i + 1);
                    scanf("%d", &age);
                    printf("Enter employee %d's salary: ", i + 1);
                    scanf("%f", &salary);
                    // Write employee data to the file
                    fprintf(fp, "%s %s %d %f\n", name, address, age, salary);
                    }
                    // Close the file
                    fclose(fp);
                    return 0;
                    }
