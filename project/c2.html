                    #include <stdio.h>
                    int main() {
                    FILE *fp;
                    int num = 123;
                    // Create a new file
                    fp = fopen("data.txt", "w");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    return 1;
                    }
                    // Write an integer to the file

                    putw(num, fp);
                    // Close the file
                    fclose(fp);
                    // Open the file again for reading
                    fp = fopen("data.txt", "r");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    return 1;
                    }
                    // Read the integer from the file
                    num = getw(fp);
                    printf("Integer read: %d\n", num);
                    // Close the file
                    fclose(fp);
                    return 0;
                    }
