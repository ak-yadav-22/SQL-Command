                #include <stdio.h>
                #include <string.h>
                int main() {
                FILE *fp;
                char str[] = "Hello, World!";
                // Create a new file  
                fp = fopen("data.txt", "w");
                if (fp == NULL) {
                printf("Error opening file!\n");
                return 1;
                }
                // Write a string to the file
                fputs(str, fp);
                // Close the file
                fclose(fp);
                // Open the file again for reading
                fp = fopen("data.txt", "r");
                if (fp == NULL) {
                printf("Error opening file!\n");
                return 1;
                }
                // Read the string from the file
                char buffer[100];
                fgets(buffer, sizeof(buffer), fp);
                printf("String read: %s\n", buffer);
                // Close the file
                fclose(fp);
                return 0;
                }
