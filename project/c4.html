                    #include <stdio.h>
                    #include <string.h>
                    int main() {
                    FILE *fp;
                    char sentence[] = "I love C programming.";
                    // Create a new file
                    fp = fopen("data.txt", "w");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    return 1;
                    }
                    // Write a sentence to the file
                    fwrite(sentence, sizeof(char), strlen(sentence), fp);
                    // Close the file
                    fclose(fp);
                    // Open the file again for reading
                    fp = fopen("data.txt", "r");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    return 1;
                    }
                    // Read the file up to 8 characters
                    char buffer[9];
                    fread(buffer, sizeof(char), 8, fp);
                    buffer[8] = '\0';
                    printf("String read: %s\n", buffer);
                    
                    // Close the file
                    fclose(fp);
                    return 0;
                    }
