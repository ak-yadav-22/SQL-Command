                    #include <stdio.h>
                    struct Employee {
                    char name[50];
                    char address[100];
                    int age;
                    float salary;
                    };
                    int main() {
                    FILE *fp;
                    struct Employee employee;
                    // Open the file for reading
                    fp = fopen("STAFF.TXT", "r");
                    if (fp == NULL) {
                    printf("Error opening file!\n");
                    
                    return 1;
                    }
                    // Read and display employee records
                    while (fscanf(fp, "%s %s %d %f", employee.name, employee.address, &employee.age,
                    &employee.salary) == 4) {
                    if (employee.salary > 15000 && employee.age < 25) {
                    printf("Name: %s\n", employee.name);
                    printf("Address: %s\n", employee.address);
                    printf("Age: %d\n", employee.age);
                    printf("Salary: %.2f\n\n", employee.salary);
                    }
                    }
                    // Close the file
                    fclose(fp);
                    return 0;
                    }
